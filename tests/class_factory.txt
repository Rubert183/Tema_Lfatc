int f(int v){
    if(v>1){
        return 1 + f(v-1);
    }
    return 0;
}

class Calculator {
    int val,
    
    int add(int v) {
        val := val + v;
        return val;
    },
    
    int getValue() {
        return val;
    },

    int f(int v) {
        if (v > 1)
        {
            return v * f(v - 1);
        }
        else {
            return 1;
        }
    }

};

Calculator createCalc(int startVal) {
    Calculator c;
    c.val := startVal;
    return c;
}
 
Calculator c1;
int v;

main {
    c1:=createCalc(7);
    Print(c1.val);
    Print(createCalc(15).add(7));
    Print(createCalc(10).val);   
    Print(createCalc(15).add(5));
    Print(c1.f(5));

    v:=10;
    while(v>1){
        v:=f(v);
        Print(v);
    }
}
