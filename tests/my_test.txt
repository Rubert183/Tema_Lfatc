class PI{
    int a,
    float get_PI(){
        return 3.14;
    }
};

class MyClass{
    int a,
    float b,
    PI p
};

MyClass new_MyClass(int a,float b){
    MyClass m_local;
    m_local.a:=a;
    m_local.b:=b;
    m_local.p.a:=10;
    return m_local;
}

MyClass m;
PI p;
float f;
main{
    f:=m.p.get_PI();
    m:=new_MyClass(1,4.12);
    Print(m.a);
    Print(new_MyClass(1,32.62).b);
    Print(f);
    Print(m.p.a);
}
